#My vim config files
## Installation
```
apt-get install vim 
cd ~
git config --global core.autocrlf input true #On Windows use proper line breaks
git clone https://github.com/Crucial2K/vim.git
mv vim .vim
mkdir bundle
cd bundle
git clone https://github.com/VundleVim/Vundle.vim
vim
:PluginInstall
```
